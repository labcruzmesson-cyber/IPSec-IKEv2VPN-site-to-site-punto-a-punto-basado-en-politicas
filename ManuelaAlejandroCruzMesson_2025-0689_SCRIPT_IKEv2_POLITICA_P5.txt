configure terminal
crypto ikev2 proposal IKEv2_PROP
 encryption aes-cbc-256
 integrity sha256
 group 14
!
crypto ikev2 policy IKEv2_POL
 proposal IKEv2_PROP
!
crypto ikev2 keyring KEYRING_PEER_B
 peer PEER_B
  address 10.0.0.70
  pre-shared-key local ClaveSeguraVpn2026
  pre-shared-key remote ClaveSeguraVpn2026
!
crypto ikev2 profile IKEv2_PROF_B
 match identity remote address 10.0.0.70 255.255.255.255
 identity local address 10.0.0.60
 authentication remote pre-share
 authentication local pre-share
 keyring local KEYRING_PEER_B
!
crypto ipsec transform-set TS_VPN esp-aes 256 esp-sha256-hmac
 mode tunnel
!
ip access-list extended ACL_VPN_TRAFFIC
 permit ip 172.16.1.0 0.0.0.255 172.16.2.0 0.0.0.255
!
ip access-list extended ACL_NAT_INTERNET
 deny ip 172.16.1.0 0.0.0.255 172.16.2.0 0.0.0.255
 permit ip 172.16.1.0 0.0.0.255 any
!
ip nat inside source list ACL_NAT_INTERNET interface GigabitEthernet0/0 overload
!
crypto map CMAP_V2 10 ipsec-isakmp
 set peer 10.0.0.70
 set transform-set TS_VPN
 set ikev2-profile IKEv2_PROF_B
 match address ACL_VPN_TRAFFIC
!
interface GigabitEthernet0/0
 ip address 10.0.0.60 255.255.255.0
 ip nat outside
 crypto map CMAP_V2
 no shutdown
!
interface GigabitEthernet0/1
 ip address 172.16.1.1 255.255.255.0
 ip nat inside
 no shutdown
!
end
write memory
